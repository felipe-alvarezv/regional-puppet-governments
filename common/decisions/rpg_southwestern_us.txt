rpg_category = {
	rpg_southwestern_us = {
		allowed = {
			NOT = {
				original_tag = USA #United States
				original_tag = D39 #Southwestern U.S.
			}
		}
		available = {
			is_subject = no
			#Oklahoma
			custom_trigger_tooltip = {
				tooltip = rpg_southwestern_us_required_states_oklahoma
				controls_state = 374 #Oklahoma
			}
			#Texas
			custom_trigger_tooltip = {
				tooltip = rpg_southwestern_us_required_states_texas
				controls_state = 375 #Texas
			}
			#New Mexico
			custom_trigger_tooltip = {
				tooltip = rpg_southwestern_us_required_states_new_mexico
				controls_state = 376 #New Mexico
			}
			#Arizona
			custom_trigger_tooltip = {
				tooltip = rpg_southwestern_us_required_states_arizona
				controls_state = 377 #Arizona
			}
		}
		visible = {
			OR = {
				controls_state = 375 #Texas
				controls_state = 376 #New Mexico
			}
			NOT = { 
                original_tag = USA #United States
				original_tag = D39 #Southwestern U.S.
                has_global_flag = rpg_southwestern_us_flag
            }
		}
		complete_effect = {
			hidden_effect = {
				374 = { add_core_of = D39 } #Oklahoma
				375 = { add_core_of = D39 } #Texas
				376 = { add_core_of = D39 } #New Mexico
				377 = { add_core_of = D39 } #Arizona
			}
			custom_effect_tooltip = rpg_southwestern_us_coring_states
			custom_effect_tooltip = rpg_southwestern_us_transfer_states
			#Set compliance to 80 in all cores and instantiate collaboration government
			custom_effect_tooltip = rpg_southwestern_us_puppet
			hidden_effect = {
				every_state = {
					limit = {
						is_core_of = D39
					}
					set_compliance = 80
				}
				
				if = {
					limit = { has_dlc = "La Resistance" }
					set_temp_variable = { country_to_initiate = D39 }
					instantiate_collaboration_government = yes
				}
				else = {
					release = D39
					puppet = D39
				}
			}
			#News event, set local flag, and set global flag
			hidden_effect = {
				ROOT = {
					set_country_flag = rpg_southwestern_us_country_flag
				}
				news_event = { id = rpg_news.19 hours = 6 }
				set_global_flag = rpg_southwestern_us_flag
			}
		}
		ai_will_do = {
			factor = 0
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
	}
}
